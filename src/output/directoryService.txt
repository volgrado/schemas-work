// src/lib/services/core/directoryService.ts

import { v4 as uuidv4 } from 'uuid';
import type { SchemaMetadata } from '$lib/types';

/**
 * Este servicio gestiona el "directorio" de todos los esquemas (documentos)
 * del usuario. Funciona como una capa de abstracción sobre localStorage para
 * manejar los metadatos de los documentos, como sus IDs, títulos y fechas.
 * No gestiona el contenido de los documentos.
 */

// Claves únicas para evitar colisiones en localStorage.
const DIRECTORY_STORAGE_KEY = 'schemas-work-directory';
const LAST_ACTIVE_DOC_KEY = 'schemas-work-last-active';

/**
 * Obtiene la lista completa de metadatos de esquemas desde localStorage.
 * Si no existe, devuelve un array vacío de forma segura.
 * Es una operación asíncrona para mantener la consistencia de la API del servicio,
 * aunque localStorage sea síncrono.
 * @returns {Promise<SchemaMetadata[]>} Una promesa que se resuelve con el array de metadatos.
 */
export async function listSchemas(): Promise<SchemaMetadata[]> {
  // En un entorno de SSR (Server-Side Rendering), `window` no existe.
  // Devolvemos un array vacío para evitar errores durante la compilación.
  if (typeof window === 'undefined') return [];

  const storedDirectory = localStorage.getItem(DIRECTORY_STORAGE_KEY);
  return storedDirectory ? JSON.parse(storedDirectory) : [];
}

/**
 * Obtiene los metadatos de un esquema específico por su ID.
 * @param {string} id - El ID del esquema a buscar.
 * @returns {Promise<SchemaMetadata | undefined>} Los metadatos del esquema o `undefined` si no se encuentra.
 */
export async function getSchemaById(
  id: string
): Promise<SchemaMetadata | undefined> {
  const schemas = await listSchemas();
  return schemas.find((schema) => schema.id === id);
}

/**
 * Obtiene el ID del último documento activo desde localStorage.
 * Permite a la aplicación reabrir el último documento en el que trabajó el usuario.
 * @returns {Promise<string | null>} El ID del último documento activo o `null`.
 */
export async function getLastActiveDocId(): Promise<string | null> {
  if (typeof window === 'undefined') return null;
  return localStorage.getItem(LAST_ACTIVE_DOC_KEY);
}

/**
 * Guarda el ID de un documento como el último activo en localStorage.
 * @param {string} id - El ID del documento a marcar como activo.
 */
export async function setLastActiveDocId(id: string): Promise<void> {
  if (typeof window === 'undefined') return;
  localStorage.setItem(LAST_ACTIVE_DOC_KEY, id);
}

/**
 * Crea los metadatos para un nuevo esquema, lo añade al directorio y persiste la lista.
 * @param {string} title - El título inicial para el nuevo esquema.
 * @returns {Promise<SchemaMetadata>} Los metadatos del esquema recién creado.
 */
export async function createSchema(title: string): Promise<SchemaMetadata> {
  const schemas = await listSchemas();
  const now = Date.now();

  const newSchema: SchemaMetadata = {
    id: uuidv4(), // Genera un identificador único universal.
    title,
    createdAt: now,
    updatedAt: now,
  };

  const updatedSchemas = [...schemas, newSchema];
  localStorage.setItem(DIRECTORY_STORAGE_KEY, JSON.stringify(updatedSchemas));

  return newSchema;
}

/**
 * Actualiza los metadatos de un esquema existente.
 * @param {string} id - El ID del esquema a actualizar.
 * @param {Partial<Omit<SchemaMetadata, 'id'>>} updates - Un objeto con los campos a actualizar (ej. { title: "Nuevo Título" }).
 * @returns {Promise<SchemaMetadata>} Los metadatos actualizados del esquema.
 * @throws {Error} Si no se encuentra un esquema con el ID proporcionado.
 */
export async function updateSchemaMetadata(
  id: string,
  updates: Partial<Omit<SchemaMetadata, 'id'>>
): Promise<SchemaMetadata> {
  const schemas = await listSchemas();
  const schemaIndex = schemas.findIndex((s) => s.id === id);

  if (schemaIndex === -1) {
    throw new Error(`No se encontró el esquema con id: ${id}`);
  }

  const updatedSchema = {
    ...schemas[schemaIndex],
    ...updates,
    updatedAt: Date.now(), // Siempre actualizamos la fecha de modificación.
  };

  schemas[schemaIndex] = updatedSchema;
  localStorage.setItem(DIRECTORY_STORAGE_KEY, JSON.stringify(schemas));

  return updatedSchema;
}

/**
 * Guarda un directorio completo de metadatos en localStorage.
 * Esta es una operación destructiva, usada principalmente por la función de importación.
 * @param {SchemaMetadata[]} schemas - El array completo de metadatos a guardar.
 */
export async function saveDirectory(schemas: SchemaMetadata[]): Promise<void> {
  if (typeof window === 'undefined') return;
  localStorage.setItem(DIRECTORY_STORAGE_KEY, JSON.stringify(schemas));
}

/**
 * Limpia por completo el directorio de metadatos de localStorage.
 * Usado por la función de importación antes de restaurar nuevos datos.
 */
export async function clearDirectory(): Promise<void> {
  if (typeof window === 'undefined') return;
  localStorage.removeItem(DIRECTORY_STORAGE_KEY);
}
